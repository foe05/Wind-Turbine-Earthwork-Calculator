# Test-DEM erstellen (500m x 500m, 1m Aufl√∂sung)

    5 * np.random.random((height, width))  # Rauschen

dataset.SetProjection(srs.ExportToWkt())

# Daten schreiben

band.WriteArray(dem_data)
band.SetNoDataValue(-9999)
